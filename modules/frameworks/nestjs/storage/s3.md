# NestJS + AWS S3

> NestJS-specific patterns for object storage via `@aws-sdk/client-s3`.
> Extends generic NestJS conventions.

## Integration Setup

```bash
npm install @aws-sdk/client-s3 @aws-sdk/s3-request-presigner multer multer-s3 uuid
npm install -D @types/multer @types/uuid
```

## S3Module (Global Provider)

```typescript
// src/s3/s3.module.ts
import { Global, Module } from '@nestjs/common';
import { S3Client } from '@aws-sdk/client-s3';
import { ConfigService } from '@nestjs/config';

export const S3_CLIENT = 'S3_CLIENT';
export const S3_BUCKET = 'S3_BUCKET';

@Global()
@Module({
  providers: [
    {
      provide: S3_CLIENT,
      inject: [ConfigService],
      useFactory: (config: ConfigService) =>
        new S3Client({
          region: config.get<string>('AWS_REGION', 'eu-central-1'),
          endpoint: config.get<string>('AWS_ENDPOINT'),         // LocalStack override
          forcePathStyle: !!config.get<string>('AWS_ENDPOINT'), // required for LocalStack
        }),
    },
    {
      provide: S3_BUCKET,
      inject: [ConfigService],
      useFactory: (config: ConfigService) => config.get<string>('S3_BUCKET'),
    },
  ],
  exports: [S3_CLIENT, S3_BUCKET],
})
export class S3Module {}
```

## StorageService

```typescript
// src/storage/storage.service.ts
import { GetObjectCommand, PutObjectCommand, DeleteObjectCommand } from '@aws-sdk/client-s3';
import { getSignedUrl } from '@aws-sdk/s3-request-presigner';

@Injectable()
export class StorageService {
  private readonly logger = new Logger(StorageService.name);

  constructor(
    @Inject(S3_CLIENT) private readonly s3: S3Client,
    @Inject(S3_BUCKET) private readonly bucket: string,
  ) {}

  async upload(key: string, body: Buffer | Readable, contentType: string): Promise<string> {
    await this.s3.send(new PutObjectCommand({
      Bucket: this.bucket,
      Key: key,
      Body: body,
      ContentType: contentType,
    }));
    this.logger.log(`Uploaded: ${key}`);
    return key;
  }

  async getPresignedUrl(key: string, expiresInSeconds = 900): Promise<string> {
    return getSignedUrl(
      this.s3,
      new GetObjectCommand({ Bucket: this.bucket, Key: key }),
      { expiresIn: expiresInSeconds },
    );
  }

  async delete(key: string): Promise<void> {
    await this.s3.send(new DeleteObjectCommand({ Bucket: this.bucket, Key: key }));
    this.logger.log(`Deleted: ${key}`);
  }
}
```

## Upload Controller with Multer

```typescript
// src/files/files.controller.ts
import { FileInterceptor } from '@nestjs/platform-express';
import multer from 'multer';
import multerS3 from 'multer-s3';

@Controller('files')
export class FilesController {
  constructor(
    @Inject(S3_CLIENT) private readonly s3: S3Client,
    @Inject(S3_BUCKET) private readonly bucket: string,
    private readonly storageService: StorageService,
  ) {}

  @Post('upload')
  @UseInterceptors(
    FileInterceptor('file', {
      storage: multerS3({
        s3: inject(S3_CLIENT),    // inject at module level in practice
        bucket: inject(S3_BUCKET),
        contentType: multerS3.AUTO_CONTENT_TYPE,
        key: (_req, file, cb) => cb(null, `uploads/${randomUUID()}/${file.originalname}`),
      }),
      limits: { fileSize: 10 * 1024 * 1024 },  // 10 MB
    }),
  )
  async uploadFile(@UploadedFile() file: Express.MulterS3.File) {
    return { key: file.key, size: file.size };
  }

  @Get(':key(*)/url')
  async getUrl(@Param('key') key: string) {
    const url = await this.storageService.getPresignedUrl(key);
    return { url };
  }

  @Delete(':key(*)')
  @HttpCode(HttpStatus.NO_CONTENT)
  async deleteFile(@Param('key') key: string): Promise<void> {
    await this.storageService.delete(key);
  }
}
```

### Configuring Multer-S3 via Module

Inject S3 client and bucket at the module level to avoid constructor injection inside the decorator:

```typescript
// files/files.module.ts
@Module({
  imports: [S3Module],
  controllers: [FilesController],
  providers: [
    StorageService,
    {
      provide: 'UPLOAD_MIDDLEWARE',
      inject: [S3_CLIENT, S3_BUCKET],
      useFactory: (s3: S3Client, bucket: string) =>
        multer({
          storage: multerS3({
            s3,
            bucket,
            contentType: multerS3.AUTO_CONTENT_TYPE,
            key: (_req, file, cb) => cb(null, `uploads/${randomUUID()}/${file.originalname}`),
          }),
          limits: { fileSize: 10 * 1024 * 1024 },
        }),
    },
  ],
})
export class FilesModule {}
```

## Scaffolder Patterns

```
src/
  s3/
    s3.module.ts             # S3Client + S3_BUCKET providers, @Global
  storage/
    storage.service.ts       # upload, getPresignedUrl, delete
    storage.module.ts
  files/
    files.module.ts
    files.controller.ts      # upload endpoint, presigned URL endpoint, delete
```

## Dos

- Make `S3Module` global — inject `S3_CLIENT` and `S3_BUCKET` tokens without re-importing the module
- Use `forcePathStyle: true` when `AWS_ENDPOINT` is set — required for LocalStack and MinIO
- Return only the object key to the client; generate presigned URLs on demand
- Validate `content-type` and enforce `limits.fileSize` before the file reaches S3
- Use `getSignedUrl` with short TTL (≤15 minutes) for download links

## Don'ts

- Don't expose raw S3 URLs — they are public if the bucket ACL is not private
- Don't buffer large file uploads in memory — stream directly using `multer-s3`
- Don't catch all S3 errors as 500 — inspect `err.$metadata.httpStatusCode` for 403/404
- Don't commit `AWS_ACCESS_KEY_ID` / `AWS_SECRET_ACCESS_KEY` — use IAM roles or environment injection
